TotalTickets = COUNTROWS(ZendeskTicketSystem)



Avg Tickets per day =
CALCULATE([TotalTickets],
('ZendeskTicketSystem'),
'ZendeskTicketSystem'[Created] > MAX('Calendar'[Date]) -1 &&
'ZendeskTicketSystem'[Created] <= MAX('Calendar'[Date])
)





Calculated Column  used for Filter

Right Click on the WksBkActual Column from ZendeskTicketSystem >>>New Column >>>Enter the syntax below.

Last 12 Weeks = Var a = MAX(ZendeskTicketSystem[WksBkActual])-10

Return

IF(ZendeskTicketSystem[WksBkActual]>= a, "Last 12 Weeks", "Other Weeks")
